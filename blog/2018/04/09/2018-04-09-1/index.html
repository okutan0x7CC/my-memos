Go でCSVファイル生成 & 書き込み

コードは以下になります。

  • リファレンス : https://golang.org/pkg/encoding/csv/


package main

import (
    "encoding/csv"
    "log"
    "os"
)



func main() {
    file, err := os.Create("./samples.csv")
    if err != nil {
        log.Println(err)
    }
    defer file.Close()

    writer := csv.NewWriter(file)
    writer.WriteAll(samples)

    writer.Flush()
}
